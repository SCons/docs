from . import ClassicCPP

SWIGSuffixes = ['.i']

def SWIGScanner():
    expr = r'^[ \t]*%[ \t]*(?:include|import|extern)[ \t]*(<|"?)([^>\s"]+)(?:>|"?)'
    scanner = ClassicCPP("SWIGScanner", ".i", "SWIGPATH", expr)
    return scanner
